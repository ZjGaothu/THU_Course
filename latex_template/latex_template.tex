    module frecounter(CLK1,rst,SIGNAL_TEST,COUT);
    input CLK1,rst,SIGNAL_TEST;
    output reg [9:0]COUT;
    reg [9:0] counter;
    reg [1:0] EDGE;
    
    always @ (posedge SIGNAL_TEST or negedge rst)
    begin
        if(rst == 0)
        begin
            counter <= 0;
            COUT <= 0;
            EDGE <= 2'b00;
        end
        else
        begin
            EDGE <= {CLK1,EDGE[1]};
            if(EDGE == 2'b10)
            begin
                COUT <= counter + 1;
                counter <= 0;
            end
            else
            begin
                COUT <= COUT;
                counter <= counter + 1;
            end
        end
    end
    endmodule
\end{lstlisting}
